020package gate.stanford;
021
022import java.util.Collection;
023
024import edu.stanford.nlp.trees.GrammaticalStructure;
025import edu.stanford.nlp.trees.GrammaticalStructure.Extras;
026import edu.stanford.nlp.trees.TypedDependency;
027
028public enum DependencyMode {
029  Typed,
030  AllTyped,
031  TypedCollapsed,
032  TypedCCprocessed;
033  
034  
035  protected static Collection<TypedDependency> getDependencies(GrammaticalStructure gs,
036      DependencyMode mode, boolean includeExtras) {
037    Collection<TypedDependency> result = null;
038    
039    Extras incl = Extras.NONE;
040    if(includeExtras) {
041      incl = Extras.MAXIMAL;
042    }
043    
044    if (mode.equals(Typed)) {
045      result = gs.typedDependencies(incl);
046    }
047    else if (mode.equals(AllTyped)) {
048      result = gs.allTypedDependencies();
049    }
050    else if (mode.equals(TypedCollapsed)) {
051      result = gs.typedDependenciesCollapsed(incl);
052    }
053    else if (mode.equals(TypedCCprocessed)) {
054      result = gs.typedDependenciesCCprocessed(incl);
055    }
056    
057    return result;
058  }
059
060}
